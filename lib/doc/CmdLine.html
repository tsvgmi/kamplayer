def self.parse_range(*input)
  result = []
  input.map{|w| w.split(',')}.flatten.each do |aword|
    case aword
    when /-/
      rstart, rend = aword.split(/-/)
      result.concat((rstart.to_i..rend.to_i).to_a)
    else
      result << aword.to_i
    end
  end
  result
end
